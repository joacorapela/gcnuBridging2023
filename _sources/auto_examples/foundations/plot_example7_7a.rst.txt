    n_samples = 10000
    n_draws = 4

    n_samples_in_event = 0
    for i in range(n_samples):
        draws = sample(n_draws=n_draws)
        if sample_in_event(sample=draws):
            n_samples_in_event += 1
    p_event = n_samples_in_event/n_samples
