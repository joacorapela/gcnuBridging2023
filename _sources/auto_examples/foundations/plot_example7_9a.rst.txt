    def sample(teams=["E1", "E2", "G1", "G2", "I1", "I2", "S1", "S2"]):
        permutation_indices = np.random.permutation(len(teams))

    n_samples_in_event = 0
    for i in range(n_samples):
        a_sample = sample()
        if sample_in_event(a_sample):

    p_event = n_samples_in_event/n_samples
